$flexy-theme-avatar: (
        default: (
                background-color: #2c3e50,
                color: #fff
        ),
        primary: (
                background-color: map-get($flexy-theme-colors, primary),
                color: #fff
        ),
        error: (
                background-color: map-get($flexy-theme-colors, error),
                color: #fff
        ),
        success: (
                background-color: map-get($flexy-theme-colors, success),
                color: #fff
        )
) !default;
